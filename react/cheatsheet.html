üìù JSX Basics

JSX is a syntax extension that looks like HTML but is actually JavaScript.

JSX Expression

// Embed JavaScript with curly braces
const name = "React";
const element = <h1>Hello, {name}!</h1>;

// Any JS expression works
<p>{2 + 2}</p>
<p>{user.firstName}</p>
<p>{formatDate(new Date())}</p>

JSX Attributes

// Use camelCase for attributes
<div className="container"></div>
<label htmlFor="name"></label>
<input tabIndex={1} />

// Spread attributes
const props = { id: "btn", disabled: true };
<button {...props}>Click</button>

Inline Styles

// Style is an object with camelCase
const styles = {
    backgroundColor: "blue",
    fontSize: "16px",
    padding: "10px"
};

<div style={styles}>Styled</div>

// Or inline
<div style={{ color: "red" }}>Red</div>

Fragments

// Group elements without extra DOM nodes
import { Fragment } from "react";

// Short syntax (most common)
<>
    <h1>Title</h1>
    <p>Content</p>
</>

// Full syntax (needed for keys)
<Fragment key={id}>
    <dt>{term}</dt>
    <dd>{definition}</dd>
</Fragment>

The building blocks of React applications.

Functional Component

// Using the component

// Usage

    return <div>Content</div>;
}

üé£ React Hooks

State and lifecycle in functional components.

useState

import { useState } from "react";

function Counter() {
    const [count, setCount] = useState(0);
    
    // Update with new value
    const increment = () => setCount(count + 1);
    
    // Update with function (safer for async)
    const incrementSafe = () => setCount(prev => prev + 1);
    
    // Update object state (spread to keep other properties)
    const [user, setUser] = useState({ name: "", age: 0 });
    const updateName = (name) => setUser(prev => ({ ...prev, name }));
    
    return <button onClick={increment}>Count: {count}</button>;
}

useEffect

import { useState, useEffect } from "react";

function DataFetcher({ userId }) {
    const [data, setData] = useState(null);
    
    // Run on every render
    useEffect(() => {
        console.log("Rendered!");
    });
    
    // Run once on mount
    useEffect(() => {
        console.log("Mounted!");
    }, []);
    
    // Run when dependency changes
    useEffect(() => {
        fetchUser(userId).then(setData);
    }, [userId]);
    
    // Cleanup function
    useEffect(() => {
        const subscription = subscribe();
        return () => subscription.unsubscribe(); // Cleanup
    }, []);
}

useRef

import { useRef } from "react";

function TextInput() {
    const inputRef = useRef(null);
    
    const focusInput = () => {
        inputRef.current.focus();
    };
    
    return (
        <>
            <input ref={inputRef} />
            <button onClick={focusInput}>
                Focus
            </button>
        </>
    );
}

// Also for mutable values
const countRef = useRef(0);
countRef.current++; // Doesn't re-render

useMemo & useCallback

import { useMemo, useCallback } from "react";

function ExpensiveComponent({ items, filter }) {
    // Memoize expensive computation
    const filtered = useMemo(() => {
        return items.filter(item => 
            item.name.includes(filter)
        );
    }, [items, filter]);
    
    // Memoize callback function
    const handleClick = useCallback((id) => {
        console.log("Clicked:", id);
    }, []); // Empty = never recreated
    
    return <List items={filtered} onClick={handleClick} />;
}

useReducer

import { useReducer } from "react";

const initialState = { count: 0 };

function reducer(state, action) {
    switch (action.type) {
        case "increment":
            return { count: state.count + 1 };
        case "decrement":
            return { count: state.count - 1 };
        default:
            throw new Error();
    }
}

function Counter() {
    const [state, dispatch] = useReducer(reducer, initialState);
    
    return (
        <>
            Count: {state.count}
            <button onClick={() => dispatch({ type: "increment" })}>+</button>
            <button onClick={() => dispatch({ type: "decrement" })}>-</button>
        </>
    );
}

Custom Hook

// Custom hooks start with "use"
function useLocalStorage(key, initialValue) {
    const [value, setValue] = useState(() => {
        const stored = localStorage.getItem(key);
        return stored ? JSON.parse(stored) : initialValue;
    });
    
    useEffect(() => {
        localStorage.setItem(key, JSON.stringify(value));
    }, [key, value]);
    
    return [value, setValue];
}

// Usage
function App() {
    const [name, setName] = useLocalStorage("name", "");
}

Key Rules

// ‚úÖ Good: Use unique ID from data
{items.map(item => (
    <Item key={item.id} {...item} />
))}

// ‚ö†Ô∏è Last resort: Index (avoid if order changes)
{items.map((item, index) => (
    <Item key={index} {...item} />
))}

// ‚ùå Bad: Random keys (causes remounting)
{items.map(item => (
    <Item key={Math.random()} {...item} />
))}

üîÄ Conditional Rendering

Showing content based on conditions.

Ternary Operator

function Greeting({ isLoggedIn }) {
    return (
        <div>
            {isLoggedIn 
                ? <UserDashboard /> 
                : <LoginForm />
            }
        </div>
    );
}

// Inline text
<p>{count === 1 ? "item" : "items"}</p>

Logical AND (&&)

function Notifications({ messages }) {
    return (
        <div>
            {messages.length > 0 && (
                <Badge count={messages.length} />
            )}
        </div>
    );
}

// ‚ö†Ô∏è Avoid with numbers (0 renders as "0")
// Use: {count > 0 && ...}
// Not: {count && ...}

Early Return

function UserProfile({ user, isLoading }) {

        return <Spinner />;

Nullish Coalescing

function Display({ value }) {
    return (
        <div>
            {/* Shows fallback if null/undefined */}
            <p>{value ?? "No value"}</p>
            
            {/* Optional chaining */}
            <p>{user?.address?.city}</p>
            
            {/* Combined */}
            <p>{user?.name ?? "Anonymous"}</p>
        </div>
    );
}

üåê Context API

Sharing data across the component tree.

Creating & Using Context

import { createContext, useContext, useState } from "react";

// 1. Create context
const ThemeContext = createContext("light");

// 2. Create provider component
function ThemeProvider({ children }) {
    const [theme, setTheme] = useState("light");
    
    const toggleTheme = () => {
        setTheme(prev => prev === "light" ? "dark" : "light");
    };
    
    return (
        <ThemeContext.Provider value={{ theme, toggleTheme }}>
            {children}
        </ThemeContext.Provider>
    );
}

// 3. Custom hook for easier access
function useTheme() {
    return useContext(ThemeContext);
}

// 4. Usage in components
function Header() {
    const { theme, toggleTheme } = useTheme();
    return (
        <header className={theme}>
            <button onClick={toggleTheme}>Toggle Theme</button>
        </header>
    );
}

// 5. Wrap app with provider
function App() {
    return (
        <ThemeProvider>
            <Header />
            <Main />
        </ThemeProvider>
    );
}

üéØ Common Patterns

Frequently used React patterns.

Controlled Input

function Form() {
    const [email, setEmail] = useState("");
    
    const handleSubmit = (e) => {
        e.preventDefault();
        console.log("Email:", email);
    };
    
    return (
        <form onSubmit={handleSubmit}>
            <input
                type="email"
                value={email}
                onChange={(e) => setEmail(e.target.value)}
            />
            <button type="submit">Submit</button>
        </form>
    );
}

Fetching Data

function UserList() {
    const [users, setUsers] = useState([]);
    const [loading, setLoading] = useState(true);
    const [error, setError] = useState(null);
    
    useEffect(() => {
        fetch("/api/users")
            .then(res => res.json())
            .then(data => {
                setUsers(data);
                setLoading(false);
            })
            .catch(err => {
                setError(err.message);
                setLoading(false);
            });
    }, []);
    
    if (loading) return <p>Loading...</p>;
    if (error) return <p>Error: {error}</p>;
    
    return <ul>{users.map(...)}</ul>;
}

Lifting State Up

function Parent() {
    const [value, setValue] = useState("");
    
    return (
        <>
            <Input value={value} onChange={setValue} />
            <Display value={value} />
        </>
    );
}

function Input({ value, onChange }) {
    return (
        <input 
            value={value} 
            onChange={(e) => onChange(e.target.value)} 
        />
    );
}

function Display({ value }) {
    return <p>You typed: {value}</p>;

Compound Components

// Tabs compound component
function Tabs({ children, defaultTab }) {
    const [active, setActive] = useState(defaultTab);
    return (
        <TabsContext.Provider value={{ active, setActive }}>
            <div className="tabs">{children}</div>
        </TabsContext.Provider>
    );
}

Tabs.List = function TabsList({ children }) {
    return <div className="tabs-list">{children}</div>;
};

Tabs.Tab = function Tab({ id, children }) {
    const { active, setActive } = useContext(TabsContext);
    return (
        <button 
            className={active === id ? "active" : ""}
            onClick={() => setActive(id)}
        >
            {children}
        </button>
    );
};

// Usage
<Tabs defaultTab="home">
    <Tabs.List>
        <Tabs.Tab id="home">Home</Tabs.Tab>
        <Tabs.Tab id="about">About</Tabs.Tab>
    </Tabs.List>
</Tabs>
