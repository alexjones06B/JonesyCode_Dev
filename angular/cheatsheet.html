‚å®Ô∏è Angular CLI Commands

Essential commands for creating and managing Angular projects.

Project Setup

# Install Angular CLI globally
npm install -g @angular/cli

# Create new project
ng new my-app

# Create with options
ng new my-app --routing --style=scss

# Serve development server
ng serve

# Build for production
ng build --configuration production

ng generate component my-component

ng g c card --inline-style
ng g c nav --inline-template

ng g module features/admin --routing

üß± Components

The building blocks of Angular applications.

Basic Component Structure

import { Component } from '@angular/core';

@Component({
    selector: 'app-hello',           // HTML tag to use
    templateUrl: './hello.component.html',  // External template
    styleUrls: ['./hello.component.css']    // External styles
})
export class HelloComponent {
    // Properties (data)
    title: string = 'Hello Angular!';
    count: number = 0;
    
    // Methods (behavior)
    increment(): void {
        this.count++;
    }
}

Inline Template & Styles

@Component({
    selector: 'app-inline',
    template: `
        <h1>{{ title }}</h1>
        <p>Count: {{ count }}</p>
    `,
    styles: [`
        h1 { color: blue; }
        p { font-size: 18px; }
    `]
})
export class InlineComponent {
    title = 'Inline Example';
    count = 0;
}

Standalone Components (Angular 14+)

import { Component } from '@angular/core';
import { CommonModule } from '@angular/common';

@Component({
    selector: 'app-standalone',
    standalone: true,
    imports: [CommonModule],
    template: `
        <p *ngIf="visible">Hello!</p>
    `
})
export class StandaloneComponent {
    visible = true;
}

Standalone components don't need NgModule declarations.

Input & Output

import { Component, Input, Output, EventEmitter } from '@angular/core';

@Component({
    selector: 'app-child',
    template: `
        <h2>{{ title }}</h2>
        <button (click)="sendMessage()">Send</button>
    `
})
export class ChildComponent {
    // Receive data from parent
    @Input() title: string = '';
    
    // Send data to parent
    @Output() message = new EventEmitter<string>();
    
    sendMessage(): void {
        this.message.emit('Hello from child!');
    }
}

üìÑ Templates

Angular template syntax for dynamic HTML.

Interpolation

<!-- Display property value -->
<h1>{{ title }}</h1>

<!-- Expressions -->
<p>Sum: {{ 1 + 1 }}</p>
<p>{{ user.name.toUpperCase() }}</p>

<!-- Method calls -->
<p>{{ getFullName() }}</p>

Use {{ }} to display dynamic values in templates.

Template Reference Variables

<!-- Reference to element -->
<input #nameInput type="text">
<button (click)="greet(nameInput.value)">
    Greet
</button>

<!-- Reference to component -->
<app-timer #timer></app-timer>
<button (click)="timer.start()">Start</button>

Safe Navigation Operator

<!-- Prevents errors if null/undefined -->
<p>{{ user?.name }}</p>
<p>{{ user?.address?.city }}</p>

<!-- Without safe navigation (may error) -->
<p>{{ user.name }}</p> <!-- ‚ùå Error if user is null -->

Use ?. to safely access properties that might be null.

Non-null Assertion

<!-- Tell Angular value is not null -->
<p>{{ user!.name }}</p>

<!-- Useful with strict null checks -->
<span>{{ data!.length }}</span>

‚ö†Ô∏è Use only when you're certain the value exists.

üîß Directives

Built-in directives for controlling DOM elements.

*ngIf - Conditional

<!-- Show/hide based on condition -->
<div *ngIf="isLoggedIn">
    Welcome back!
</div>

<!-- With else -->
<div *ngIf="isLoggedIn; else loginPrompt">
    Dashboard
</div>
<ng-template #loginPrompt>
    Please log in
</ng-template>

<!-- With then/else -->
<div *ngIf="loading; then spinner else content"></div>
<ng-template #spinner>Loading...</ng-template>
<ng-template #content>Data loaded!</ng-template>

*ngFor - Loop

<!-- Basic loop -->
<ul>
    <li *ngFor="let item of items">
        {{ item }}
    </li>
</ul>

<!-- With index -->
<li *ngFor="let item of items; let i = index">
    {{ i }}: {{ item }}
</li>

<!-- With trackBy for performance -->
<li *ngFor="let user of users; trackBy: trackByUserId">
    {{ user.name }}
</li>

// In component:
trackByUserId(index: number, user: User): number {
    return user.id;
}

*ngSwitch - Multiple Conditions

<div [ngSwitch]="status">
    <p *ngSwitchCase="'loading'">
        Loading...
    </p>
    <p *ngSwitchCase="'success'">
        Data loaded!
    </p>
    <p *ngSwitchCase="'error'">
        Error occurred
    </p>
    <p *ngSwitchDefault>
        Unknown status
    </p>
</div>

ngClass & ngStyle

<!-- Dynamic classes -->
<div [ngClass]="{'active': isActive, 'disabled': isDisabled}">
</div>

<!-- Array of classes -->
<div [ngClass]="['btn', 'btn-primary']"></div>

<!-- Dynamic styles -->
<div [ngStyle]="{
    'color': textColor,
    'font-size.px': fontSize
}">
    Styled text
</div>

@if / @for (Angular 17+)

<!-- New control flow syntax -->

@if (isLoggedIn) {
    <p>Welcome!</p>
} @else {
    <p>Please log in</p>
}

@for (item of items; track item.id) {
    <li>{{ item.name }}</li>
} @empty {
    <li>No items found</li>
}

@switch (status) {
    @case ('loading') { <spinner /> }
    @case ('error') { <error /> }
    @default { <content /> }
}

New built-in control flow available in Angular 17+

üîó Data Binding

Connecting component data to templates.

Binding Overview

<!-- Interpolation: Component ‚Üí Template -->
<h1>{{ title }}</h1>

<!-- Property Binding: Component ‚Üí Template -->
<img [src]="imageUrl">
<button [disabled]="isDisabled">Click</button>

<!-- Event Binding: Template ‚Üí Component -->
<button (click)="handleClick()">Click</button>
<input (input)="onInput($event)">

<!-- Two-way Binding: Component ‚Üî Template -->
<input [(ngModel)]="username">

<!-- Attribute Binding -->
<td [attr.colspan]="colSpan">Cell</td>

<!-- Class Binding -->
<div [class.active]="isActive"></div>

<!-- Style Binding -->
<div [style.color]="textColor"></div>

Event Binding Details

<!-- Click event -->
<button (click)="onClick()">Click</button>

<!-- With $event -->
<input (input)="onInput($event)">

<!-- Keyboard events -->
<input (keyup.enter)="onEnter()">
<input (keydown.escape)="onEscape()">

<!-- Mouse events -->
<div (mouseenter)="onHover()">
<div (mouseleave)="onLeave()">

Two-way Binding

<!-- Requires FormsModule -->
import { FormsModule } from '@angular/forms';

<!-- In template -->
<input [(ngModel)]="name">
<p>Hello, {{ name }}!</p>

<!-- Expanded syntax (banana in a box) -->
<input 
    [ngModel]="name" 
    (ngModelChange)="name = $event">

[()] is called "banana in a box" syntax!

‚öôÔ∏è Services & Dependency Injection

Sharing logic and data across components.

Creating a Service

import { Injectable } from '@angular/core';

@Injectable({
    providedIn: 'root'  // Available app-wide
})
export class UserService {
    private users: User[] = [];
    
    getUsers(): User[] {
        return this.users;
    }
    
    addUser(user: User): void {
        this.users.push(user);
    }
    
    getUserById(id: number): User | undefined {
        return this.users.find(u => u.id === id);
    }
}

Injecting a Service

import { Component } from '@angular/core';
import { UserService } from './user.service';

@Component({
    selector: 'app-user-list',
    template: `
        <ul>
            <li *ngFor="let user of users">
                {{ user.name }}
            </li>
        </ul>
    `
})
export class UserListComponent {
    users: User[];
    
    // Inject via constructor
    constructor(private userService: UserService) {
        this.users = this.userService.getUsers();
    }
}

inject() Function (Angular 14+)

import { Component, inject } from '@angular/core';
import { UserService } from './user.service';

@Component({...})
export class UserListComponent {
    // Inject using inject() function
    private userService = inject(UserService);
    
    users = this.userService.getUsers();
}

Modern alternative to constructor injection.

HTTP Service

import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';

@Injectable({ providedIn: 'root' })
export class ApiService {
    private baseUrl = 'https://api.example.com';
    
    constructor(private http: HttpClient) {}
    
    getUsers(): Observable<User[]> {
        return this.http.get<User[]>(`${this.baseUrl}/users`);
    }
    
    createUser(user: User): Observable<User> {
        return this.http.post<User>(`${this.baseUrl}/users`, user);
    }
}

üîÑ Lifecycle Hooks

React to component lifecycle events.

All Lifecycle Hooks

import { 
    Component, OnInit, OnDestroy, OnChanges, 
    AfterViewInit, SimpleChanges 
} from '@angular/core';

@Component({...})
export class LifecycleComponent implements OnInit, OnDestroy, OnChanges, AfterViewInit {
    
    // Called once when @Input properties change (before ngOnInit)
    ngOnChanges(changes: SimpleChanges): void {
        console.log('Input changed:', changes);
    }
    
    // Called once after first ngOnChanges (component initialized)
    ngOnInit(): void {
        console.log('Component initialized');
        // Good for: fetching data, setting up subscriptions
    }
    
    // Called after view (and child views) initialized
    ngAfterViewInit(): void {
        console.log('View initialized');
        // Good for: accessing @ViewChild elements
    }
    
    // Called when component is destroyed
    ngOnDestroy(): void {
        console.log('Component destroyed');
        // Good for: cleanup, unsubscribing
    }
}

Common Usage Pattern

import { Component, OnInit, OnDestroy } from '@angular/core';
import { Subscription } from 'rxjs';

@Component({...})
export class DataComponent implements OnInit, OnDestroy {
    private subscription?: Subscription;
    data: any[] = [];
    
    constructor(private dataService: DataService) {}
    
    ngOnInit(): void {
        // Subscribe to data
        this.subscription = this.dataService
            .getData()
            .subscribe(data => this.data = data);
    }
    
    ngOnDestroy(): void {
        // Prevent memory leaks
        this.subscription?.unsubscribe();
    }
}

Lifecycle Order

// Order of lifecycle hook execution:

1. constructor()      // DI happens here
2. ngOnChanges()      // Input bindings change
3. ngOnInit()         // Component initialized
4. ngDoCheck()        // Change detection runs
5. ngAfterContentInit()   // Content projected
6. ngAfterContentChecked()
7. ngAfterViewInit()  // View initialized
8. ngAfterViewChecked()
9. ngOnDestroy()      // Cleanup before destroy

Most commonly used: ngOnInit, ngOnDestroy, ngOnChanges

üß≠ Routing

Navigate between views in your application.

Route Configuration

// app-routing.module.ts
import { NgModule } from '@angular/core';
import { RouterModule, Routes } from '@angular/router';

const routes: Routes = [
    { path: '', component: HomeComponent },
    { path: 'about', component: AboutComponent },

    imports: [RouterModule.forRoot(routes)],
    exports: [RouterModule]
})
export class AppRoutingModule {}

import { Router, ActivatedRoute } from '@angular/router';

@Component({...})
export class NavComponent {
    constructor(
        private router: Router,
        private route: ActivatedRoute
    ) {}
    
    goToUser(id: number): void {
        this.router.navigate(['/users', id]);
    }
    
    goHome(): void {
        this.router.navigateByUrl('/');
    }

Route Parameters

import { ActivatedRoute } from '@angular/router';

@Component({...})
export class UserDetailComponent implements OnInit {
    userId!: string;
    
    constructor(private route: ActivatedRoute) {}
    
    ngOnInit(): void {
        // Snapshot (one-time)
        this.userId = this.route.snapshot.paramMap.get('id')!;
        
        // Observable (reactive)
        this.route.paramMap.subscribe(params => {
            this.userId = params.get('id')!;
        });
    }
}

import { FormGroup, FormControl, Validators } from '@angular/forms';

@Component({...})

    }

üîß Pipes

Transform data for display in templates.

Built-in Pipes

<!-- Text transformation -->
{{ name | uppercase }}         <!-- ALEX -->
{{ name | lowercase }}         <!-- alex -->
{{ name | titlecase }}         <!-- Alex -->

<!-- Number formatting -->
{{ price | number }}           <!-- 1,234.57 -->
{{ price | number:'1.0-2' }}   <!-- 1,234.57 -->
{{ ratio | percent }}          <!-- 85% -->
{{ price | currency }}         <!-- $1,234.57 -->
{{ price | currency:'EUR' }}   <!-- ‚Ç¨1,234.57 -->

<!-- Date formatting -->
{{ today | date }}             <!-- Dec 15, 2025 -->
{{ today | date:'short' }}     <!-- 12/15/25 -->
{{ today | date:'fullDate' }}  <!-- Monday, Dec 15, 2025 -->
{{ today | date:'HH:mm' }}     <!-- 14:30 -->

Object & Array Pipes

<!-- JSON display -->
<pre>{{ user | json }}</pre>

<!-- Array slice -->
<li *ngFor="let item of items | slice:0:5">
    {{ item }}
</li>

<!-- Async pipe (Observables) -->
<div *ngIf="user$ | async as user">
    {{ user.name }}
</div>

<!-- KeyValue pipe (objects) -->
<div *ngFor="let item of obj | keyvalue">
    {{ item.key }}: {{ item.value }}
</div>

Custom Pipe

import { Pipe, PipeTransform } from '@angular/core';

@Pipe({
    name: 'truncate'
})
export class TruncatePipe implements PipeTransform {
    transform(value: string, limit: number = 50, ellipsis: string = '...'): string {
        if (!value) return '';
        if (value.length <= limit) return value;
        return value.substring(0, limit) + ellipsis;
    }
}

// Usage in template:
{{ longText | truncate:20 }}       // First 20 chars + ...
{{ longText | truncate:30:'‚Üí' }}   // First 30 chars + ‚Üí

Common Gotchas

// ‚ùå Don't subscribe in templates like this:
{{ getData() | async }}  // Called every change detection!

// ‚úÖ Store observable in property:
data$ = this.service.getData();
// {{ data$ | async }}

// ‚ùå Don't forget to unsubscribe:
this.sub = observable.subscribe();
// Memory leak!

// ‚úÖ Unsubscribe in ngOnDestroy:
ngOnDestroy() { this.sub.unsubscribe(); }

// ‚úÖ Or use async pipe (auto-unsubscribes)
// ‚úÖ Or use takeUntilDestroyed()
